Here are some lines of code I used for this sketch.

let pic, dotcolour;

function preload() {
pic=loadImage("data/Stamps+Me.jpg");
}

function setup() {
createCanvas(windowWidth, windowHeight);
pic.resize(width, 0);
}

function draw() {
let x = random(width);
let y = random(height);
noStroke();
dotcolour=pic.get(x, y);
fill(dotcolour);
circle(x, y, random(20,80));
}
