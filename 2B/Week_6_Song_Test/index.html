Here are some lines of code I used for this sketch.

var song;
var analyzer;

function preload() {
song = loadSound ('data/zero-gravity.wav');
}

function setup() {
createCanvas (windowWidth, windowHeight);
background(0);
analyzer = new p5.Amplitude();
analyzer.setInput(song);
fill(120);
textSize(72);
textAlign(CENTER,CENTER);
}

function draw() {
background(252, 148, 124, 10);
var volume = analyzer.getLevel();
volume*=300;
fill(16, 13, 2);
textSize(volume*1.5);
translate(width/2,height/2);
textX=map(volume, 0, 400, 0, windowWidth);
textY=map(volume, 0, 400, 0, windowHeight);
rotate(volume);
text('do you like my song?', 0, 0);
}

function mousePressed() {
if (song.isPlaying()) {
song.stop();
song.noLoop();
}
else {
song.loop();
song.play();
}
}
